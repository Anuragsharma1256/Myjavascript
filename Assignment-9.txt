// find the area of a triangle.  Lengths of its sides are 5, 6, 7


// 
const a = 5;
const b = 6;
const c = 7;

const s = (a + b + c) / 2;
const area = Math.sqrt(s * (s - a) * (s - b) * (s - c));

console.log("The area of the triangle is:", area);
